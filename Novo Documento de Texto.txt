----------------------- INICIANDO -----------------------
1 - mkdir siteProjeto
2 - python -m venv dsweb
3 - .\dsweb\Scripts\activate
4 - pip install django; pip list
5 - django-admin startproject projeto_nome // dir raiz + manage.py
6 - cd mysite; python manage.py runserver // "roda o servidor"
7 - python manage.py startapp app_nome // cria uma app
